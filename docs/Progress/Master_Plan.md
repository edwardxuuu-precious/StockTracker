from backtrader import Strategy

class TestStrategy(Strategy):
    def __init__(self):
        self.sma = bt.indicators.SMA(period=20)

    def next(self):
        if self.data.close > self.sma:
            self.buy()
        elif self.data.close < self.sma:
            self.sell()

# 运行回测
cerebro = bt.Cerebro()
cerebro.addstrategy(TestStrategy)
cerebro.run()
